from zipfile import ZipFile
import os

# Create a new folder
folder_path = "/mnt/data/Better_GPS_GoogleMap_Viewer"
os.makedirs(folder_path, exist_ok=True)

# Improved HTML content (no auto-refresh or trail, but better appearance)
html_content = '''<!DOCTYPE html>
<html>
<head>
  <title>Live GPS Tracker</title>
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <style>
    html, body {
      height: 100%;
      margin: 0;
      font-family: Arial, sans-serif;
    }
    #map {
      height: 100%;
      width: 100%;
    }
    header {
      background-color: #4285F4;
      color: white;
      padding: 10px 20px;
      font-size: 1.5em;
      text-align: center;
    }
  </style>
</head>
<body>
  <header>Live GPS Tracker - ThingSpeak to Google Maps</header>
  <div id="map"></div>

  <script>
    const channelID = 3007161;
    const latField = 1;
    const lonField = 2;

    async function getGPSData() {
      const response = await fetch(`https://api.thingspeak.com/channels/${channelID}/feeds.json?results=1`);
      const data = await response.json();
      const feed = data.feeds[0];
      return {
        lat: parseFloat(feed[`field${latField}`]),
        lon: parseFloat(feed[`field${lonField}`])
      };
    }

    async function initMap() {
      const { lat, lon } = await getGPSData();

      const map = new google.maps.Map(document.getElementById("map"), {
        zoom: 16,
        center: { lat, lng: lon },
        mapTypeControl: false,
        fullscreenControl: true,
        streetViewControl: false
      });

      new google.maps.Marker({
        position: { lat, lng: lon },
        map,
        title: "Live Location",
        animation: google.maps.Animation.DROP
      });
    }
  </script>

  <script async defer
    src="https://maps.googleapis.com/maps/api/js?key=AIzaSyC3DTNe4Ql9JkPMPKoplQBGRfjGjSo1SmE&callback=initMap">
  </script>
</body>
</html>'''

# Save HTML file
html_file_path = os.path.join(folder_path, "index.html")
with open(html_file_path, "w") as f:
    f.write(html_content)

# Create a ZIP
zip_path = "/mnt/data/Better_GPS_GoogleMap_Viewer.zip"
with ZipFile(zip_path, 'w') as zipf:
    zipf.write(html_file_path, arcname="index.html")

zip_path
